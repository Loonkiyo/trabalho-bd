CREATE OR ALTER PROCEDURE Incluir_Pessoa
    @Nome VARCHAR(50),
    @Endereco VARCHAR(50),
    @Cidade VARCHAR(40),
    @Email NVARCHAR(40),
    @Data_Nasc DATETIME
AS
BEGIN

IF NOT EXISTS (SELECT 1 FROM Pessoa WHERE Email = @Email)

    INSERT INTO Pessoa (Nome, Endereco, Cidade, Email, Data_Nasc)
    VALUES (@Nome, @Endereco, @Cidade, @Email, @Data_Nasc)

ELSE
    PRINT 'Pessoa já cadastrada'

END



#ADICIONAR PESSOA
